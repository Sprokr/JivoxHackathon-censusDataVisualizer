CREATE TABLE IF NOT EXISTS educationData (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL DEFAULT "DefaultName",
    age INT(3) NOT NULL DEFAULT 0,
    gender VARCHAR(20) NOT NULL DEFAULT "UNCLASSIFIED",
    studies VARCHAR(20) NOT NULL DEFAULT "UNCLASSIFIED",
    currentClass VARCHAR(20) NOT NULL DEFAULT "UNCLASSIFIED",
    lastClass VARCHAR(20) NOT NULL DEFAULT "UNCLASSIFIED",
    stateCode INT(2) NOT NULL DEFAULT 0,
    income INT(12) NOT NULL DEFAULT 0,
    PRIMARY KEY(id)
);

CREATE TABLE IF NOT EXISTS stateCode (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL DEFAULT "DefaultName",
    PRIMARY KEY(id)
);
